// Arduino Code Snippet: Obstacle Detection & Collision Avoidance

#include <Arduino.h>
#define trigPin 9
#define echoPin 10
#define motorLeftForward 3
#define motorLeftBackward 4
#define motorRightForward 5
#define motorRightBackward 6

long duration;
int distance;

void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  pinMode(motorLeftForward, OUTPUT);
  pinMode(motorLeftBackward, OUTPUT);
  pinMode(motorRightForward, OUTPUT);
  pinMode(motorRightBackward, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  // Measure distance
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  duration = pulseIn(echoPin, HIGH);
  distance = duration * 0.034 / 2;

  Serial.println(distance);

  if(distance < 15) {
    // Obstacle detected, move backward and turn
    digitalWrite(motorLeftBackward, HIGH);
    digitalWrite(motorRightBackward, HIGH);
    delay(500);
    digitalWrite(motorLeftBackward, LOW);
    digitalWrite(motorRightBackward, LOW);
    // Turn
    digitalWrite(motorLeftForward, HIGH);
    digitalWrite(motorRightBackward, HIGH);
    delay(500);
    digitalWrite(motorLeftForward, LOW);
    digitalWrite(motorRightBackward, LOW);
  } else {
    // Move forward
    digitalWrite(motorLeftForward, HIGH);
    digitalWrite(motorRightForward, HIGH);
  }
}
